/**
 * Rozšíření DOM
 */ 
JSLab.dom.extend = (function(){

	var test = function(){};
	var E = window.HTMLElement || window.Element; // v IE není HTMLElement ale jen Element

	E.prototype.__proto_test__ = test; // rozšíření o testovací fci.
	var div = document.createElement("div");

	if(div.__proto_test__ === test){
		// rozšíření nativního objektu funguje
		JSLab.extend(E.prototype, JSLab.dom.methods);
		// dom.extend tedy nemusí dělat nic
		return function(element){
			return element;
		};
	}else{
		// je třeba rozšiřovat každý element samostatně
		return function (element){
			return JSLab.extend(element, JSLab.dom.methods);
		};
	}

	// úklid
	delete E.prototype.__proto_test__;
	delete div;

})();
